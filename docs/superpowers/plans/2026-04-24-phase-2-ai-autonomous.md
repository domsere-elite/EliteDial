# Phase 2 — AI Autonomous Worker + Compliance Preamble Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the AI Autonomous outbound worker for EliteDial: a concurrency-capped, compliance-gated dial loop that places SignalWire calls whose SWML connect-leg bridges to a per-campaign Retell agent's SIP endpoint.

**Architecture:** Five new building blocks — `ConcurrencyLimiter` (process-local semaphore behind an interface), `DialPrecheck` (TCPA + DNC + Reg F composer), `AIAutonomousWorker` (event-driven + 30 s safety-net interval), an `eventBus` (typed Node EventEmitter), and `envValidation` (fail-fast at boot). The SWML builder gains a Retell-SIP-bridge variant; the SignalWire service generalises its SWML query plumbing.

**Tech Stack:** Node 20+, TypeScript strict mode (build), Express 4, Prisma 5.x, Zod, `node:test`, no Redis, no BullMQ.

**Source spec:** [docs/superpowers/specs/2026-04-24-phase-2-ai-autonomous-design.md](../specs/2026-04-24-phase-2-ai-autonomous-design.md). All architectural decisions are locked there. Do not re-litigate.

```typescript
import { prisma } from '../lib/prisma';
import { logger } from '../utils/logger';

export interface ConcurrencyLimiter {
    acquire(campaignId: string, cap: number): boolean;
    release(campaignId: string): void;
    active(campaignId: string): number;
    rebuildFromDb(): Promise<void>;
    sweepStuck(): number;
}

export interface ConcurrencyLimiterDeps {
    rebuildSource: () => Promise<Map<string, number>>;
    clock?: () => number;
    stuckTimeoutMs?: number;
}

export function buildProcessLocalLimiter(deps: ConcurrencyLimiterDeps): ConcurrencyLimiter {
    const counts = new Map<string, number>();
    // Per-slot acquire timestamps for stuck-slot detection. Stored as a flat list per campaign;
    // length should always equal counts.get(campaignId).
    const acquiredAt = new Map<string, number[]>();
    const clock = deps.clock || Date.now;
    const stuckTimeoutMs = deps.stuckTimeoutMs ?? 600_000;

    return {
        acquire(campaignId, cap) {
            const current = counts.get(campaignId) || 0;
            if (current >= cap) return false;
            counts.set(campaignId, current + 1);
            const stamps = acquiredAt.get(campaignId) || [];
            stamps.push(clock());
            acquiredAt.set(campaignId, stamps);
            return true;
        },
        release(campaignId) {
            const current = counts.get(campaignId) || 0;
            if (current <= 0) return;
            counts.set(campaignId, current - 1);
            const stamps = acquiredAt.get(campaignId) || [];
            stamps.shift();
            acquiredAt.set(campaignId, stamps);
        },
        active(campaignId) {
            return counts.get(campaignId) || 0;
        },
        async rebuildFromDb() {
            const seeded = await deps.rebuildSource();
            counts.clear();
            acquiredAt.clear();
            const now = clock();
            for (const [id, n] of seeded.entries()) {
                counts.set(id, n);
                acquiredAt.set(id, new Array(n).fill(now));
            }
        },
        sweepStuck() {
            const now = clock();
            let released = 0;
            for (const [id, stamps] of acquiredAt.entries()) {
                const surviving = stamps.filter(t => now - t < stuckTimeoutMs);
                const removed = stamps.length - surviving.length;
                if (removed > 0) {
                    released += removed;
                    acquiredAt.set(id, surviving);
                    counts.set(id, Math.max(0, (counts.get(id) || 0) - removed));
                    logger.warn('concurrency-limiter: released stuck slots', { campaignId: id, count: removed });
                }
            }
            return released;
        },
    };
}

// Default rebuild source: counts active ai_outbound calls grouped by campaignId via CampaignAttempt.
const defaultRebuildSource = async (): Promise<Map<string, number>> => {
    const rows = await prisma.campaignAttempt.findMany({
        where: {
            call: {
                status: { in: ['initiated', 'ringing', 'in-progress'] },
                mode: 'ai_outbound',
            },
        },
        select: { campaignId: true },
    });
    const counts = new Map<string, number>();
    for (const r of rows) {
        counts.set(r.campaignId, (counts.get(r.campaignId) || 0) + 1);
    }
    return counts;
};

export const concurrencyLimiter: ConcurrencyLimiter = buildProcessLocalLimiter({
    rebuildSource: defaultRebuildSource,
});
```

```typescript
/**
 * Compute the next 8 AM in the given timezone, relative to `now`.
 * If we are currently before 8 AM local, returns 8 AM today; otherwise 8 AM tomorrow.
 */
export const nextCallingWindowStart = (timezone?: string | null, now: Date = new Date()): Date => {
    const tz = timezone || DEFAULT_TIMEZONE;
    const formatter = new Intl.DateTimeFormat('en-US', {
        timeZone: tz,
        hour: 'numeric',
        hour12: false,
    });
    let localHour: number;
    try {
        localHour = parseInt(formatter.format(now), 10);
    } catch {
        localHour = parseInt(
            new Intl.DateTimeFormat('en-US', { timeZone: DEFAULT_TIMEZONE, hour: 'numeric', hour12: false }).format(now),
            10,
        );
    }
    // Days to add: 0 if before 8 AM, 1 otherwise.
    const dayOffset = localHour < CALLING_WINDOW_START_HOUR ? 0 : 1;
    // Build target as YYYY-MM-DDT08:00 in the contact tz, then convert via Date constructor.
    // We approximate by adding (dayOffset * 86_400_000) ms and snapping to 8:00 local-tz hour.
    const base = new Date(now.getTime() + dayOffset * 86_400_000);
    const dateParts = new Intl.DateTimeFormat('en-CA', {
        timeZone: tz,
        year: 'numeric', month: '2-digit', day: '2-digit',
    }).formatToParts(base);
    const y = dateParts.find(p => p.type === 'year')!.value;
    const m = dateParts.find(p => p.type === 'month')!.value;
    const d = dateParts.find(p => p.type === 'day')!.value;
    // Construct the wall-clock 8:00 in tz; rely on Date's parsing of an ISO with no offset assumed UTC,
    // then correct by subtracting the tz offset for that wall-clock moment.
    const utcGuess = new Date(`${y}-${m}-${d}T${String(CALLING_WINDOW_START_HOUR).padStart(2, '0')}:00:00Z`);
    const tzGuessHour = parseInt(
        new Intl.DateTimeFormat('en-US', { timeZone: tz, hour: 'numeric', hour12: false }).format(utcGuess),
        10,
    );
    const offsetHours = CALLING_WINDOW_START_HOUR - tzGuessHour;
    return new Date(utcGuess.getTime() + offsetHours * 3_600_000);
};
```

```typescript
import type { Campaign, CampaignContact } from '@prisma/client';
import { dncService } from './dnc';
import { complianceFrequency, RegFCheckResult } from './compliance-frequency';
import { isWithinCallingWindow, nextCallingWindowStart, getContactTimezone } from './tcpa';
import { logger } from '../utils/logger';

export interface DialPrecheckResult {
    allowed: boolean;
    blockedReasons: string[];
    deferUntil?: Date;
}

export interface DialPrecheckDeps {
    tcpa: {
        isWithinCallingWindow(tz: string | null): boolean;
        nextCallingWindowStart(tz: string | null, now?: Date): Date;
    };
    dnc: { isOnDNC(phone: string): Promise<boolean> };
    regF: { checkRegF(phone: string): Promise<RegFCheckResult> };
    clock?: () => Date;
}

export interface DialPrecheck {
    precheck(campaign: Pick<Campaign, 'timezone'>, contact: Pick<CampaignContact, 'primaryPhone' | 'timezone'>): Promise<DialPrecheckResult>;
}

export function buildDialPrecheck(deps: DialPrecheckDeps): DialPrecheck {
    const clock = deps.clock || (() => new Date());
    return {
        async precheck(campaign, contact) {
            const reasons: string[] = [];
            let deferUntil: Date | undefined;

            const tz = getContactTimezone(contact.timezone, campaign.timezone);

            // 1. TCPA quiet hours
            if (!deps.tcpa.isWithinCallingWindow(tz)) {
                reasons.push('tcpa_quiet_hours');
                deferUntil = deps.tcpa.nextCallingWindowStart(tz, clock());
            }

            // 2. DNC
            try {
                if (await deps.dnc.isOnDNC(contact.primaryPhone)) {
                    reasons.push('dnc_listed');
                }
            } catch (err) {
                logger.error('dial-precheck: DNC check threw — fail-safe block', { err });
                reasons.push('dnc_check_failed');
            }

            // 3. Reg F
            try {
                const r = await deps.regF.checkRegF(contact.primaryPhone);
                if (r.blocked) reasons.push('reg_f_cap');
            } catch (err) {
                logger.error('dial-precheck: RegF check threw — fail-safe block', { err });
                reasons.push('reg_f_check_failed');
            }

            return {
                allowed: reasons.length === 0,
                blockedReasons: reasons,
                ...(deferUntil ? { deferUntil } : {}),
            };
        },
    };
}

export const dialPrecheck: DialPrecheck = buildDialPrecheck({
    tcpa: { isWithinCallingWindow, nextCallingWindowStart },
    dnc: { isOnDNC: (p) => dncService.isOnDNC(p) },
    regF: { checkRegF: (p) => complianceFrequency.checkRegF(p) },
});
```

```typescript
import { EventEmitter } from 'node:events';

export type EventMap = {
    'call.terminal':       { callId: string; signalwireCallId: string; campaignId: string | null; status: string };
    'campaign.activated':  { campaignId: string };
    'campaign.paused':     { campaignId: string };
};

type EventName = keyof EventMap;

export interface EventBus {
    on<E extends EventName>(event: E, listener: (payload: EventMap[E]) => void): void;
    off<E extends EventName>(event: E, listener: (payload: EventMap[E]) => void): void;
    emit<E extends EventName>(event: E, payload: EventMap[E]): void;
}

export function buildEventBus(): EventBus {
    const emitter = new EventEmitter();
    emitter.setMaxListeners(50);
    return {
        on: (event, listener) => { emitter.on(event, listener as (...args: any[]) => void); },
        off: (event, listener) => { emitter.off(event, listener as (...args: any[]) => void); },
        emit: (event, payload) => { emitter.emit(event, payload); },
    };
}

export const eventBus: EventBus = buildEventBus();
```

```typescript
import { config } from '../config';
import { prisma } from './prisma';
import { logger } from '../utils/logger';

export interface BootEnvSnapshot {
    signalwire: { projectId: string; apiToken: string; spaceUrl: string };
    retell: { apiKey: string };
}

export interface BootEnvCheck {
    ok: boolean;
    errors: string[];
    warnings: string[];
}

export function checkBootEnv(snap: BootEnvSnapshot): BootEnvCheck {
    const errors: string[] = [];
    const warnings: string[] = [];

    const swSet = [snap.signalwire.projectId, snap.signalwire.apiToken, snap.signalwire.spaceUrl]
        .filter(v => v && v.length > 0).length;
    if (swSet > 0 && swSet < 3) {
        errors.push(
            'SIGNALWIRE_PROJECT_ID, SIGNALWIRE_API_TOKEN, and SIGNALWIRE_SPACE_URL must all be set together (live mode) or all be empty (mock mode). Partial config is rejected.',
        );
    }

    if (swSet === 3 && !snap.retell.apiKey) {
        warnings.push(
            'SignalWire is configured but RETELL_API_KEY is not. ai_autonomous campaigns will be skipped by the worker until Retell is configured.',
        );
    }

    return { ok: errors.length === 0, errors, warnings };
}

export function validateEnvOrExit(): void {
    const result = checkBootEnv({
        signalwire: {
            projectId: config.signalwire.projectId,
            apiToken: config.signalwire.apiToken,
            spaceUrl: config.signalwire.spaceUrl,
        },
        retell: { apiKey: config.retell.apiKey },
    });
    for (const w of result.warnings) logger.warn(w);
    if (!result.ok) {
        for (const e of result.errors) logger.error(e);
        // eslint-disable-next-line no-process-exit
        process.exit(1);
    }
}

export async function validateActivationsOrWarn(): Promise<void> {

                { retellSipAddress: null },
                { retellAgentPromptVersion: null },
            ],
        },
        select: { id: true, name: true },
    });

        logger.error('Active ai_autonomous campaign missing required Retell config — worker will skip it', {
```

```typescript
import { ConcurrencyLimiter, concurrencyLimiter } from './concurrency-limiter';
import { DialPrecheck, DialPrecheckResult, dialPrecheck } from './dial-precheck';
import { EventBus, eventBus } from '../lib/event-bus';
import { campaignReservationService } from './campaign-reservation-service';
import { signalwireService } from './signalwire';
import { callSessionService } from './call-session-service';
import { didRouter } from './did-router';
import { config } from '../config';
import { prisma } from '../lib/prisma';
import { logger } from '../utils/logger';

type CampaignSlim = {
    id: string;
    dialMode: string;
    status: string;
    maxConcurrentCalls: number;
    retryDelaySeconds: number;
    timezone: string;
    retellAgentId: string | null;
    retellSipAddress: string | null;
    retellAgentPromptVersion: string | null;
};

type ReserveResult = {
    contact: { id: string; primaryPhone: string; timezone: string | null };
    reservationToken: string;
};

export interface AIAutonomousWorkerDeps {
    limiter: ConcurrencyLimiter;
    eventBus: EventBus;
    precheck: DialPrecheck;
    loadCampaign: (id: string) => Promise<CampaignSlim | null>;
    listActiveAiCampaigns: () => Promise<{ id: string }[]>;
    reserveNext: (campaign: CampaignSlim) => Promise<ReserveResult | null>;
    confirmDial: (contactId: string, token: string) => Promise<void>;
    failReservation: (contactId: string, status: 'queued' | 'failed', nextAttemptAt: Date | null) => Promise<void>;
    applyBlockedStatus: (contactId: string, pre: DialPrecheckResult) => Promise<void>;
    writeBlockedCallRow: (campaign: CampaignSlim, contact: ReserveResult['contact'], reasons: string[]) => Promise<void>;
    writeInitiatedCallRow: (campaign: CampaignSlim, contact: ReserveResult['contact'], result: { provider: string; providerCallId: string }) => Promise<void>;
    initiateCall: (req: {
        fromNumber: string; toNumber: string; callbackUrl: string; swmlQuery: Record<string, string>; metadata?: Record<string, unknown>;
    }) => Promise<{ provider: string; providerCallId: string } | null>;
    pickDid: (campaign: CampaignSlim, contact: ReserveResult['contact']) => Promise<string | null>;
    callbackUrl: string;
    intervalMs?: number;
    clock?: () => Date;
}

export interface AIAutonomousWorker {
    start(): Promise<void>;
    stop(): void;
    tick(campaignId: string): Promise<void>;
}

export function buildAIAutonomousWorker(deps: AIAutonomousWorkerDeps): AIAutonomousWorker {
    const intervalMs = deps.intervalMs ?? 30_000;
    const clock = deps.clock || (() => new Date());
    let interval: NodeJS.Timeout | null = null;
    const inFlight = new Map<string, Promise<void>>();
    const loggedSkip = new Set<string>();

    const onCallTerminal = (p: { campaignId: string | null }) => {
        if (!p.campaignId) return;
        deps.limiter.release(p.campaignId);
        void worker.tick(p.campaignId);
    };
    const onCampaignActivated = (p: { campaignId: string }) => { void worker.tick(p.campaignId); };
    const onCampaignPaused = (p: { campaignId: string }) => {
        // No active dispatch to abort — in-flight calls drain via webhooks. Just clear the skip-log.
        loggedSkip.delete(p.campaignId);
    };

    const worker: AIAutonomousWorker = {
        async start() {
            await deps.limiter.rebuildFromDb();
            deps.eventBus.on('call.terminal', onCallTerminal);
            deps.eventBus.on('campaign.activated', onCampaignActivated);
            deps.eventBus.on('campaign.paused', onCampaignPaused);
            interval = setInterval(async () => {
                try {
                    const campaigns = await deps.listActiveAiCampaigns();
                    for (const c of campaigns) await worker.tick(c.id);
                    deps.limiter.sweepStuck();
                } catch (err) {
                    logger.error('ai-autonomous-worker: interval tick failed', { err });
                }
            }, intervalMs);
            logger.info('ai-autonomous-worker: started', { intervalMs });
        },
        stop() {
            if (interval) clearInterval(interval);
            interval = null;
            deps.eventBus.off('call.terminal', onCallTerminal);
            deps.eventBus.off('campaign.activated', onCampaignActivated);
            deps.eventBus.off('campaign.paused', onCampaignPaused);
            logger.info('ai-autonomous-worker: stopped');
        },
        async tick(campaignId: string) {
            const existing = inFlight.get(campaignId);
            if (existing) {
                await existing;
                return;
            }
            const p = (async () => {
                try { await runOnce(campaignId); } finally { inFlight.delete(campaignId); }
            })();
            inFlight.set(campaignId, p);
            await p;
        },
    };

    async function runOnce(campaignId: string): Promise<void> {
        const campaign = await deps.loadCampaign(campaignId);
        if (!campaign || campaign.status !== 'active' || campaign.dialMode !== 'ai_autonomous') return;
        if (!campaign.retellAgentId || !campaign.retellSipAddress || !campaign.retellAgentPromptVersion) {
            if (!loggedSkip.has(campaignId)) {
                logger.warn('ai-autonomous-worker: skipping campaign — missing Retell config', { campaignId });
                loggedSkip.add(campaignId);
            }
            return;
        }
        loggedSkip.delete(campaignId);

        if (campaign.maxConcurrentCalls <= 0) {
            if (!loggedSkip.has(`${campaignId}:noconc`)) {
                logger.warn('ai-autonomous-worker: skipping — maxConcurrentCalls <= 0', { campaignId });
                loggedSkip.add(`${campaignId}:noconc`);
            }
            return;
        }

        // eslint-disable-next-line no-constant-condition
        while (true) {
            if (deps.limiter.active(campaignId) >= campaign.maxConcurrentCalls) break;
            const reservation = await deps.reserveNext(campaign);
            if (!reservation) break;

            const pre = await deps.precheck.precheck(campaign, reservation.contact);
            if (!pre.allowed) {
                await deps.writeBlockedCallRow(campaign, reservation.contact, pre.blockedReasons);
                await deps.applyBlockedStatus(reservation.contact.id, pre);
                continue;
            }

            if (!deps.limiter.acquire(campaignId, campaign.maxConcurrentCalls)) break;

            try {
                await deps.confirmDial(reservation.contact.id, reservation.reservationToken);
                const did = await deps.pickDid(campaign, reservation.contact);
                if (!did) throw new Error('no_did_available');
                const result = await deps.initiateCall({
                    fromNumber: did,
                    toNumber: reservation.contact.primaryPhone,
                    callbackUrl: deps.callbackUrl,
                    swmlQuery: { mode: 'ai_autonomous', campaignId, from: did },
                    metadata: { campaignId, contactId: reservation.contact.id, mode: 'ai_autonomous' },
                });
                if (!result) throw new Error('signalwire_initiate_returned_null');
                await deps.writeInitiatedCallRow(campaign, reservation.contact, result);
            } catch (err) {
                deps.limiter.release(campaignId);
                const retryAt = new Date(clock().getTime() + Math.max(30, campaign.retryDelaySeconds) * 1000);
                await deps.failReservation(reservation.contact.id, 'queued', retryAt);
                logger.error('ai-autonomous-worker: dial failed', { campaignId, contactId: reservation.contact.id, err });
            }
        }
    }

    return worker;
}
```
